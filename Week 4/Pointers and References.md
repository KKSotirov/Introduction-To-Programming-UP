**1. address of operator - \& - взема адрес**



Пример: int a = 5;

&nbsp;	cout << \&a; ---> показва адреса на а



**2. value of operator - \* (за pointer-и = дереференциране) - взема стойност**



Пример: int x = 5;

&nbsp;	int \*ptr = \&x;



*//pointer, на който не знаем типът, към който сочи --> void \*ptr = \&x; (приема всякакви стойности) ((трудно се извежда))*



&nbsp;	за примера --> x = 5 си има адрес (да кажем) 104

&nbsp;	оттам pointer-a ни също си има собствен адрес, като да кажем е 204

&nbsp;	стойността в клетката на адрес 104 в паметта = 5, а стойността на адрес 204 е х, като х е още 5



&nbsp;	cout << (\*ptr) // извежда стойността на адресът на ptr, тоест стойността на х, тоест 5
